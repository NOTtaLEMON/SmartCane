Figure 13 — Firmware Algorithm (Pseudocode)

Single-loop, non-blocking control flow.

init():
    Serial.begin(115200)
    Wire.begin(SDA=21, SCL=22, clk=100 kHz)
    Serial2.begin(115200, RX=16, TX=17)        # TF-Luna
    vl53.init() · mpu.wake() · lcd.init()
    wifi.connect(SSID, PSK)
    ws.begin(port=81)

loop():                                          # every LOOP_MS = 80 ms
    fwd  = readLidarCm()                          # 0–800 cm
    drop = readTofMm()                            # 0–2000 mm
    aMag = sqrt(ax² + ay² + az²)                  # raw IMU

    if aMag > FALL_THRESHOLD (26000):
        fallAlertActive = true
        fallStartTime   = now()
    if fallAlertActive and now() − fallStartTime > 5000 ms:
        fallAlertActive = false

    fallFlag = fallAlertActive ? 1 : 0

    pkt = sprintf("%03d,%03d,%d", fwd, drop, fallFlag)
    Serial.println(pkt)
    ws.broadcastTXT(pkt)

    if  fallFlag:                buzzer(CONTINUOUS) · led(ON)
    elif fwd ≤ 50:                buzzer(BEEP_FAST)
    elif fwd ≤ 100:               buzzer(BEEP_MEDIUM)
    elif fwd ≤ 150:               buzzer(BEEP_SLOW)
    elif drop > 400 mm:           buzzer(DROP_WARN)
    else:                          buzzer(OFF) · led(OFF)

    lcd.update(fwd, drop, fallFlag)

Figure 14 — TTS Dispatch Algorithm (Pseudocode)

Two-frame streak gate prevents flicker between labels.

state:
    lastTopLabel  : String? = null
    topLabelStreak: Int     = 0
    lastSpokenAt  : Map[label → millis] = {}

on_frame(detections):
    merged = mainCocoDets + potholeDets + poleDets + treeDets + carDets + stepsDets
    merged = nms(merged, IoU=0.45)
    merged = filter(d → d.conf ≥ thresholdFor(d.label))   # per-class

    drawOverlay(merged)

    if merged.isEmpty():
        lastTopLabel   = null
        topLabelStreak = 0
        return

    top = merged.maxBy(d → d.conf)

    if top.label == lastTopLabel:
        topLabelStreak += 1
    else:
        lastTopLabel   = top.label
        topLabelStreak = 1

    if topLabelStreak ≥ 2
       and top.conf ≥ TTS_MIN_CONFIDENCE (0.60)
       and now() − lastSpokenAt[top.label] > ALERT_COOLDOWN_MS (1000):
        tts.speak(friendlyLabel(top.label))
        lastSpokenAt[top.label] = now()
